def buy_series_train(list):
    Settings.MoveMouseDelay = 0.1
    for train in list:
        main_reg.exists(, 0.5)
        reg2 = main_reg.find().right(175)
        reg2.doubleClick()
        type(Key.DELETE)
        type(train+Key.TAB)
        wait(0.1)
        buy_two_ticket = Pattern().similar(0.45).targetOffset(0,-30)
        if not reg2.below(100).exists(, 0) :
            if reg2.right(300).exists(, 0):
                reg2.below(100).click(Pattern().targetOffset(26,2))
                reg2.below(300).exists(buy_two_ticket, 0.5)
                reg2.below(300).click(buy_two_ticket)
            elif reg2.right(300).exists(, 0):
                reg2.below(100).click(Pattern().targetOffset(26,2))
                reg2.below(300).exists(buy_two_ticket, 0.5)
                reg2.below(300).click(buy_two_ticket)
            else:
                reg2.below(100).click(Pattern().similar(0.48))
                reg2.below(300).exists(buy_two_ticket, 0.5)
                reg2.below(300).click(buy_two_ticket)
        reg2.below(300).click()

        sec_reg.exists(,0.5)
        sleep(0.05)
        pic = sec_reg.find()
        key = input( u"The 驗證" )
        sec_reg.paste(Pattern().targetOffset(-49,-13), key)
        sec_reg.click()
        wait(0.2)
        if third_reg.exists(, 0.5) or third_reg.exists(, 0.0) \
                or third_reg.exists(, 0):
            type(Key.BACKSPACE)
            wait(0.2)
            third_reg.click()
        else :
            break
    Settings.MoveMouseDelay = 0.5
